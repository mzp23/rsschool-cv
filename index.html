function isValidIP(str) {
const regEx = /\n| |e/g;
if (regEx.test(str)) return false;
let ip = str.split(".");
if (ip.length < 4 || ip.length > 4) return false;
let valid = ip.every(
item => item && (item >= 0 && item <= 255) && !item.match(/^0\d{1,2}/)
);
return valid;
}
